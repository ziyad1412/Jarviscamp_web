CREATE TABLE users(id INT AUTO_INCREMENT PRIMARY KEY,name VARCHAR(255),email VARCHAR(255),password VARCHAR(255));
categories =
CREATE TABLE categories(id INT AUTO_INCREMENT PRIMARY KEY,name VARCHAR(255));
items =
CREATE TABLE items(id INT AUTO_INCREMENT PRIMARY KEY,category_id INT,name VARCHAR(255),description VARCHAR(255),price INT,stock INT,FOREIGN KEY(category_id) REFERENCES categories(id) );
